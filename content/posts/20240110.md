## React.useState()
{{< highlight react >}}
export default function App() {
    const [count, changeCount] = React.useState(0)
    function add() {
        changeCount(count - 1)
    }
    function subtract() {
        changeCount(count + 1)
    }
    return (
        <button onClick={subtract}>-</button>
            <div><h1>{count}</h1></div>
        <button onClick={add}>+</button>
    )
}
{{< /highlight >}}
